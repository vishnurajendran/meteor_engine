22 // Converts a single argument to std::string, respecting an optional
23 // format specifier string (the part after ':' in {N:spec}).
24 //
25 // Supported specifiers:
26 // fN — fixed-point float with N decimal places e.g. {0:f2}
27 // x — lowercase hexadecimal integer e.g. {0:x}
28 // X — uppercase hexadecimal integer e.g. {0:X}
29 // eN — scientific notation with N decimal places e.g. {0:e3}
30 // 0N — zero-padded integer to width N e.g. {0:04}
31 // + — show sign for positive numbers e.g. {0:+}
32 // (empty) — default std::to_string / SString::str()
33
34 template<typename T>
35 static std::string applySpec(T&& val, const std::string& spec)
36 {
37 using D = std::decay_t<T>;
38
39 // SString passthrough
40 if constexpr (std::is_same_v<D, SString>)
41 return val.coreStr;
42
43 // std::string passthrough
44 else if constexpr (std::is_same_v<D, std::string>)
45 return val;
46
47 // const char* passthrough
48 else if constexpr (std::is_convertible_v<D, const char*>)
49 return std::string(val);
50
51 // bool
52 else if constexpr (std::is_same_v<D, bool>)
53 return val ? "true" : "false";
54
55 // Numeric types — inspect specifier
56 else if constexpr (std::is_arithmetic_v<D>)
57 {
58 if (spec.empty())
59 return std::to_string(val);
60
61 std::ostringstream oss;
62
63 // f → fixed float, optional digit count: "f2", "f4" …
64 if (spec[0] == 'f')
65 {
66 int prec = (spec.size() > 1) ? std::stoi(spec.substr(1)) : 6;
67 oss << std::fixed << std::setprecision(prec) << val;
68 return oss.str();
69 }
70
71 // e → scientific notation: "e3" …
72 if (spec[0] == 'e' || spec[0] == 'E')
73 {
74 int prec = (spec.size() > 1) ? std::stoi(spec.substr(1)) : 6;
75 oss << (spec[0] == 'E' ? std::uppercase : std::nouppercase)
76 << std::scientific << std::setprecision(prec) << val;
77 return oss.str();
78 }
79
80 // x / X → hexadecimal integer
81 if (spec[0] == 'x' || spec[0] == 'X')
82 {
83 if constexpr (std::is_integral_v<D>)
84 {
85 oss << (spec[0] == 'X' ? std::uppercase : std::nouppercase)
86 << std::hex << val;
87 return oss.str();
88 }
89 return std::to_string(val);
90 }
91
92 // 0N → zero-padded integer width: "04", "08" …
93 if (spec[0] == '0' && spec.size() > 1)
94 {
95 int width = std::stoi(spec.substr(1));
96 oss << std::setfill('0') << std::setw(width) << val;
97 return oss.str();
98 }
99
100 // + → force sign
101 if (spec[0] == '+')
102 {
103 oss << std::showpos << val;
104 return oss.str();
105 }
106
107 return std::to_string(val);
108 }
109
110 // Fallback: try streaming into ostringstream
111 else
112 {
113 std::ostringstream oss;
114 oss << val;
115 return oss.str();
116 }
117 }

186 // ── C#-style formatting ───────────────────────────────────────────────────
187 //
188 // Usage:
189 // SString::format("Hello {0}, you scored {1:f2} points!", "Alice", 98.765f)
190 // → "Hello Alice, you scored 98.77 points!"
191 //
192 // SString::format("0x{0:X} ({0:04})", 255)
193 // → "0xFF (0255)" (same index can appear multiple times)
194 //
195 // Specifiers after ':' (optional):
196 // fN — fixed float, N decimal places {0:f2}
197 // eN — scientific notation N decimals {0:e3}
198 // x — lowercase hex {0:x}
199 // X — uppercase hex {0:X}
200 // 0N — zero-padded width N {0:04}
201 // + — force sign {0:+}
202
203 template<typename... Args>
204 static SString format(const SString& fmt, Args&&... args)
205 {
206 // Store one converter lambda per argument.
207 // Each lambda closes over a copy of its argument and applies the
208 // format specifier on demand, so the same index can appear multiple
209 // times in the format string with different specifiers.
210 //
211 // We use the std::initializer_list expansion trick instead of a fold
212 // expression because GCC cannot always see the pack inside a lambda
213 // capture when writing (push_back([v = fwd(args)](...){...}), ...).
214 std::vector<std::function<std::string(const std::string&)>> converters;
215 converters.reserve(sizeof...(Args));
216 (void)std::initializer_list<int>{
217 (converters.push_back(
218 [v = std::forward<Args>(args)](const std::string& spec) mutable -> std::string {
219 return applySpec(v, spec);
220 }
221 ), 0)...
222 };
223
224 std::string result = fmt.coreStr;
225 std::string output;
226 output.reserve(result.size());
227
228 size_t i = 0;
229 while (i < result.size())
230 {
231 // Escaped brace: {{ → {, }} → }
232 if (result[i] == '{' && i + 1 < result.size() && result[i+1] == '{')
233 { output += '{'; i += 2; continue; }
234 if (result[i] == '}' && i + 1 < result.size() && result[i+1] == '}')
235 { output += '}'; i += 2; continue; }
236
237 // Format token: {N} or {N:spec}
238 if (result[i] == '{')
239 {
240 size_t close = result.find('}', i + 1);
241 if (close == std::string::npos)
242 throw std::invalid_argument("SString::format — unclosed '{' in format string");
243
244 std::string token = result.substr(i + 1, close - i - 1);
245 std::string spec;
246 size_t colon = token.find(':');
247 if (colon != std::string::npos)
248 {
249 spec = token.substr(colon + 1);
250 token = token.substr(0, colon);
251 }
252
253 size_t idx = static_cast<size_t>(std::stoul(token));
254 if (idx >= converters.size())
255 throw std::out_of_range("SString::format — argument index out of range");
256
257 output += converters[idx](spec);
258 i = close + 1;
259 continue;
260 }
261
262 output += result[i++];
263 }
264
265 return SString(output);
266 }
