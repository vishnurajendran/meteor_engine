17// The pipeline collects these once in preRender(); each stage decides
18// independently what to do with them (bind material, skip it, etc.).
19//
20// Geometry contract:
21// ebo != 0 → glDrawElements using indexCount
22// ebo == 0 → glDrawArrays using vertexCount
23//
24// vao/ebo are GL object handles (GLuint = unsigned int). This header
25// intentionally does not include any GL headers — include glew.h in
26// the .cpp files that actually issue draw calls.

38
39 // Material — may be nullptr.
40 // Depth stage ignores it; opaque stage calls material->bindMaterial().
41 MMaterial* material = nullptr;

44 // Set per-entity via MStaticMeshEntity::setCastsShadow().
