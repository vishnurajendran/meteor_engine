2// Suggested location: core/engine/assetmanagement/asset/tassethandle.h
3//
4// A stable, typed reference to an asset. Stores the asset's GUID rather than
5// a raw pointer, so it survives both:
6// - Full MAssetManager::refresh() (old pointers are deleted and replaced)
7// - In-place requestReload() (the MAsset* itself stays alive)
8//
9// ── Migration ────────────────────────────────────────────────────────────────
10// Before: MTextureAsset* tex = manager->getAsset<MTextureAsset>(path);
11// After: TAssetHandle<MTextureAsset> tex = manager->getAsset<MTextureAsset>(path);
12//
13// You can also wrap a raw pointer you already have:
14// TAssetHandle<MShaderAsset> handle(existingRawPtr);
15//
16// ── Access ────────────────────────────────────────────────────────────────────
17// tex->getTexture() — operator-> resolves through the manager
18// tex.get() — explicit resolve, may return nullptr
19// if (tex) { ... } — bool cast, false when asset not found
20// ─────────────────────────────────────────────────────────────────────────────
21

30template<typename T>
31class TAssetHandle
32{
33 static_assert(std::is_base_of_v<MAsset, T>, "T must inherit from MAsset");
34 SString assetId;
35
36public:
37 TAssetHandle() = default;
38
39 explicit TAssetHandle(const SString& id)
40 : assetId(id) {}
41
42 // Implicit conversion from raw pointer — convenience for migration.
43 // The handle will be null if the pointer has no assetId set yet (i.e.
44 // the meta file hasn't been written). That only happens in tests; in
45 // normal editor flow every asset gets a GUID on first load.
46 TAssetHandle(T* asset) // NOLINT(google-explicit-constructor)
47 : assetId(asset ? asset->getAssetId() : SString()) {}
48
49 // Resolves through the manager every call. O(log n) map lookup.
50 // Returns nullptr if the asset isn't loaded or the handle is null.
51 T* get() const
52 {
53 if (assetId.empty())
54 return nullptr;
55 return MAssetManager::getInstance()->getAssetById<T>(assetId);
56 }
57
58 T* operator->() const { return get(); }
59 T& operator*() const { return *get(); }
60
61 // True if the handle has a GUID AND the asset is currently loaded.
62 explicit operator bool() const { return isValid(); }
63 bool isValid() const { return get() != nullptr; }
64
65 // True if the handle was never assigned (no GUID at all).
66 bool isNull() const { return assetId.empty(); }
67
68 const SString& getAssetId() const { return assetId; }
69
70 bool operator==(const TAssetHandle<T>& other) const { return assetId == other.assetId; }
71 bool operator!=(const TAssetHandle<T>& other) const { return assetId != other.assetId; }
72};

74#endif // ASSETHANDLE_H
