14// TAssetRef<T>
15//
16// A serializable, typed reference to an asset. Stores both the asset's GUID
17// and its file path so it can be persisted to XML (via the Field system) and
18// lazily resolved at runtime.
19//
20// Resolution order:
21// 1. Try GUID through MAssetManager::getAssetById<T>().
22// 2. If that fails and a path is present, fall back to
23// MAssetManager::getAsset<T>(path).
24// 3. On a successful path-based fallback, backfill the GUID so subsequent
25// resolves take the fast path.
26//
27// This ensures that post-copy directories and older scene files that only
28// contain asset paths keep working, while new saves always emit the GUID
29// for rename-safe references.
30//
31// Usage in an entity:
32//
33// DECLARE_FIELD(meshAsset, TAssetRef<MStaticMeshAsset>, {})
34//
35// // runtime access
36// if (auto* mesh = meshAsset.get().resolve())
37// mesh->getMeshes();
38//
39// // assignment from a handle or raw pointer
40// meshAsset = someHandle;
41// meshAsset = someRawPtr;
42//
43

112 // -- resolution ----------------------------------------------------------

114 // Returns a TAssetHandle built from the current GUID.
115 // Does NOT include the path fallback. Use resolve() for full resolution.
116 TAssetHandle<T> getHandle() const
117 {
118 const auto am = MEngineSubsystemRegistry::getSubsystem<IAssetManagerSubsystem>();
119 return am->getAssetById<T>(m_assetId);
120 }

122 // Full resolution with fallback.
123 // Tries GUID first, then path. Backfills the GUID on a successful
124 // path-based resolve so subsequent calls take the fast path.
125 // O(log n) map lookup per attempt.
126 T* resolve() const
127 {
128 auto* manager = MEngineSubsystemRegistry::getSubsystem<IAssetManagerSubsystem>();
129 if (!manager)
130 return nullptr;
131
132 // Fast path: resolve by GUID.
133 if (!m_assetId.empty())
134 {
135 auto handle = manager->getAssetById<T>(m_assetId);
136 if (T* asset = handle.get())
137 return asset;
138 }
139
140 // Fallback: resolve by file path.
141 if (!m_path.empty())
142 {
143 auto handle = manager->getAsset<T>(m_path);
144 if (T* asset = handle.get())
145 {
146 // Backfill the GUID so future resolves skip the fallback.
147 m_assetId = asset->getAssetId();
148 return asset;
149 }
150 }
151
152 return nullptr;
153 }
