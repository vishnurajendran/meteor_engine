4// Asynchronous waveform peak generator for editor audio previews.
5// Decodes the audio file on a background thread at its native channel
6// count, reduces to min/max peaks per bin per channel, and exposes
7// the result for ImGui rendering.
8//
9// Usage:
10// auto* wf = MAudioWaveform::generate("assets/audio/clip.mp3", 512);
11// // ... each frame ...
12// if (wf->isReady()) { /* read peaks per channel */ }
13// // ... when done ...
