33 void registerType(const MTypeInfo& info, FactoryFn factory)
34 {
35 if (byHash.count(info.hash))

37 std::cerr << "[EntityTypeRegistry] Duplicate type: '" << info.name << "'\n";

40 byHash[info.hash] = std::move(factory);

50 // Create by class name string (used when reading XML)
51 // Converts name -> hash once, then does a hash lookup — still O(1).
52 //
53 MSpatialEntity* create(const std::string& typeName) const
54 {
55 auto it = nameToHash.find(typeName);
56 if (it == nameToHash.end())
57 {
58 MERROR(SString::format("[EntityTypeRegistry] Unknown type: '{0}'",typeName));
59 return nullptr;
60 }
61 return createByHash(it->second);
62 }
63
64 // Query
65 [[nodiscard]] bool hasType(const MTypeInfo& info) const
66 {
67 return byHash.count(info.hash) > 0;
68 }
69
70 [[nodiscard]] bool hasType(const std::string& name) const
