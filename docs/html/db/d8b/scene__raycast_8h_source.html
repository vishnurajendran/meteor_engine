34 // Möller–Trumbore ray-triangle intersection.
35 // Returns true if the ray (origin + t*dir) hits the triangle v0-v1-v2.
36 // On hit, outT is set to the distance along the ray.
37 inline bool rayTriangle(const SVector3& origin, const SVector3& dir,
38 const SVector3& v0, const SVector3& v1, const SVector3& v2,
39 float& outT)
40 {
41 SVector3 e1 = v1 - v0;
42 SVector3 e2 = v2 - v0;
43 SVector3 h = glm::cross(dir, e2);
44 float a = glm::dot(e1, h);
45 if (std::abs(a) < 1e-7f) return false;
46
47 float f = 1.0f / a;
48 SVector3 s = origin - v0;
49 float u = f * glm::dot(s, h);
50 if (u < 0.0f || u > 1.0f) return false;
51
52 SVector3 q = glm::cross(s, e1);
53 float v = f * glm::dot(dir, q);
54 if (v < 0.0f || u + v > 1.0f) return false;
55
56 outT = f * glm::dot(e2, q);
57 return outT > 1e-6f;
58 }
59
60 // Ray vs Y=0 ground plane. Returns true if the ray hits the plane.
61 inline bool rayGroundPlane(const SVector3& origin, const SVector3& dir,
62 float& outT)
63 {
64 if (std::abs(dir.y) < 1e-7f) return false;
65 outT = -origin.y / dir.y;
66 return outT > 0.0f;
67 }
