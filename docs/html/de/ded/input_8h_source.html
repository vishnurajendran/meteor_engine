63class SInput
64{
65public:
66 // Snapshot the current keyboard state from the OS. Call exactly once per
67 // frame, before any pressedThisFrame / pressedAndReleased / isDown queries.
68 static void poll();
69
70 // Rising edge -- the key transitioned from up to down this frame.
71 static bool pressedThisFrame(EKeyCode key);
72
73 // Falling edge -- the key transitioned from down to up this frame
74 // (i.e. the user completed a press-and-release gesture).
75 static bool pressedAndReleased(EKeyCode key);
76
77 // Raw held state -- the key is physically down right now.
78 static bool isDown(EKeyCode key);
79
80 // Modifier convenience helpers (left OR right counts).
81 static bool isCtrlDown();

83 static bool isAltDown();
84
85 // Human-readable name for a key code (e.g. EKeyCode::S -> "S").
86 static const char* getKeyName(EKeyCode key);

88private:
