2// field_asset_ref_types.h

4// Field<T> partial specialization for TAssetRef<AssetType>.
5//
6// Include this in any header that uses DECLARE_FIELD with a TAssetRef,
7// AFTER both data/field.h and asset_ref_handle.h are visible.
8// Follows the same pattern as field_engine_types.h.
9//
10// Serializes both the asset GUID and file path into XML. On load, reads
11// whichever is present. This supports three scenarios:
12// - New files: both <id> and <path> are present.
13// - Legacy files: only <path> is present (pre-GUID scene files).
14// - Copied directories: <path> is valid, <id> may not resolve yet.
15//
16// XML format:
17// <fieldName>
18// <id>guid-string</id>
19// <path>assets/meshes/cube.obj</path>
20// </fieldName>
21//

45 // Write both GUID and path as child nodes.
46 void write(pugi::xml_node& parent) const override
47 {
48 auto node = parent.append_child(name.c_str());
49 node.append_child("id").text().set(rawValue.getAssetId().c_str());
50 node.append_child("path").text().set(rawValue.getPath().c_str());
51 }
52
53 // Read whichever children are present. Missing nodes are silently
54 // ignored, so legacy files with only <path> still load correctly.
55 void load(const pugi::xml_node& parent) override
56 {
57 auto node = parent.child(name.c_str());
58 if (!node) return;
59
60 if (auto idNode = node.child("id"))
61 rawValue.setAssetId(SString(idNode.text().as_string("")));
62
63 if (auto pathNode = node.child("path"))
64 rawValue.setPath(SString(pathNode.text().as_string("")));
65 }
66};
