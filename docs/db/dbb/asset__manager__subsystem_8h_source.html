20 // Every handle returned resolves by GUID. The path is only used to
21 // locate the asset and extract its GUID at creation time.
22 template <typename T>
23 TAssetHandle<T> getAsset(SString path) {
24 static_assert(std::is_base_of_v<MAsset, T>, "T must inherit from MAsset");
25
26 if (assetMap.contains(path)) {
27 auto* raw = dynamic_cast<T*>(assetMap[path]);
28 if (raw && !raw->getAssetId().empty())
29 return TAssetHandle<T>(raw->getAssetId(), &resolveByGuid);
30 }
31 return TAssetHandle<T>();
32 }
33
34 template <typename T>
35 TAssetHandle<T> getAssetById(const SString& assetId) {
36 static_assert(std::is_base_of_v<MAsset, T>, "T must inherit from MAsset");
37
38 if (assetMapByAssetId.contains(assetId))
39 return TAssetHandle<T>(assetId, &resolveByGuid);
40 return TAssetHandle<T>();
41 }

43 // Parses the "guid::<id>" / "rawp::<path>" serialisation format.
44 //
45 // "guid::abc-123" : getAssetById<T>("abc-123")
46 // "rawp::assets/tex.png" : getAsset<T>("assets/tex.png")
47 //
48 // Both return a GUID handle. On re-save, toRefString() writes "guid::".
49 // This lets scene files use human-readable paths as defaults that are
50 // automatically upgraded to stable GUIDs once the asset is loaded.
51
52 template <typename T>
53 TAssetHandle<T> getAssetFromReference(const SString& refString) {
54 static_assert(std::is_base_of_v<MAsset, T>, "T must inherit from MAsset");
55
56 const std::string& s = refString.str();
57
58 if (s.size() > REF_PREFIX_LEN)
59 {
60 if (s.compare(0, REF_PREFIX_LEN, REF_GUID_PREFIX) == 0)
61 return getAssetById<T>(SString(s.substr(REF_PREFIX_LEN)));
62
63 if (s.compare(0, REF_PREFIX_LEN, REF_PATH_PREFIX) == 0)
64 return getAsset<T>(SString(s.substr(REF_PREFIX_LEN)));
65 }
66 return TAssetHandle<T>();
67 }
