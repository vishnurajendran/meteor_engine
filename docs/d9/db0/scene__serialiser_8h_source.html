4// Saves and loads an entire MScene to/from XML.
5//
6// Usage:
7// MSceneSerializer::save(scene, "levels/level01.scene");
8// MSceneSerializer::load("levels/level01.scene", scene);
9//
10// Scene file format:
11//
12// <scene name="Level01">
13// <entity type="spatial" name="Floor" enabled="true">
14// <relativePosition><x>0</x><y>0</y><z>0</z></relativePosition>
15// ...
16// <children>
17// <entity type="point_light" name="LightA">
18// <intensity>2.5</intensity>
19// ...
20// </entity>
21// </children>
22// </entity>
23// <entity type="camera" name="MainCamera">
24// ...
25// </entity>
26// </scene>
27//
28// Only ROOT entities are written at the top level; children are nested under
29// their parent's <children> block. The full hierarchy is preserved exactly.
30//
