60 static SShaderPropertyType parsePropertyType(const SString& str)
61 {
62 if (str.toLower() == TYPE_INT) return Int;
63 if (str.toLower() == TYPE_FLOAT) return Float;
64 if (str.toLower() == TYPE_BOOL) return Bool;

67 if (str.toLower() == TYPE_FLOAT4) return UniformVec4;
68 if (str.toLower() == TYPE_COLOR) return Color;
69 if (str.toLower() == TYPE_MATRIX) return Matrix4;
70 if (str.toLower() == TYPE_TEXTURE2D) return Texture;
71 if (str.toLower() == TYPE_TEXTURE) return Texture;
