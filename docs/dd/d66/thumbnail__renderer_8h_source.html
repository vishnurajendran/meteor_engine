1// thumbnailrenderer.h
2//
3// Renders asset previews into a shared 128×128 SFrameBuffer, reads the pixels
4// back with glReadPixels, and stores the result as an sf::Texture in
5// MThumbnailCache. One request is processed per tick() call so the editor
6// frame time stays predictable.
7//
8// ── Material ──────────────────────────────────────────────────────────────────
9// Renders a UV sphere with the material's colour and texture applied via the
10// built-in 3-point Lambert shader. Textures are mapped using spherical UVs
11// derived from the surface normal — no vertex UV attribute needed.
12//
13// ── MStaticMesh requirements ─────────────────────────────────────────────────
14// MThumbnailRenderer calls these methods on MStaticMesh:
15//
16// unsigned int getVAO() const;
17// unsigned int getEBO() const;
18// int getIndexCount() const;
19// int getVertexCount()const;
20// AABB getBounds() const;
