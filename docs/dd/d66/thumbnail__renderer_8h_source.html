1// thumbnailrenderer.h
2// Suggested location: editor/editorassetmanager/thumbnails/
3//
4// ── What this does ────────────────────────────────────────────────────────────
5// Renders asset previews into a shared 128×128 SFrameBuffer, reads the pixels
6// back with glReadPixels, and stores the result as an sf::Texture in
7// MThumbnailCache. One request is processed per tick() call so the editor
8// frame time stays predictable.
9//
10// ── Material ──────────────────────────────────────────────────────────────────
11// Renders a UV sphere with the material's colour properties applied to the
12// built-in 3-point Lambert shader. No .mesl dependency needed.
13//
14// ── MStaticMesh requirements ─────────────────────────────────────────────────
15// MThumbnailRenderer calls these methods on MStaticMesh. Add them if missing:
16//
17// unsigned int getVAO() const; // GL vertex array object
18// unsigned int getEBO() const; // GL element buffer (0 if none)
19// int getIndexCount() const; // number of indices
20// int getVertexCount()const; // number of vertices (if no EBO)
21// AABB getBounds() const; // world-space bounding box
22//
23// The AABB is used to fit the camera to the mesh. If getBounds() isn't
24// available yet, the renderer falls back to a unit-sphere camera distance.
25// ─────────────────────────────────────────────────────────────────────────────

100 // computeThumbnailCamera() calculates the view and projection matrices
