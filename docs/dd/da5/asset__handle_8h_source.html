1// asset_handle.h
2//
3// A stable, typed reference to an asset. Stores the asset's GUID and a
4// resolve callback. The handle always resolves by GUID — it does not know
5// about file paths or the asset manager.
6//
7// ── Usage ─────────────────────────────────────────────────────────────────────
8// TAssetHandle<MTextureAsset> tex = manager->getAsset<MTextureAsset>(path);
9// tex->getTexture() — operator-> resolves via callback + dynamic_cast
10// tex.get() — explicit resolve, may return nullptr
11// if (tex) { ... } — bool cast, false when asset not found
12//
13// ── Converting between types ──────────────────────────────────────────────────
14// TAssetHandle<MAsset> base = manager->getAsset<MAsset>(path);
15// TAssetHandle<MStaticMeshAsset> mesh = base; // downcast — safe at runtime
16// TAssetHandle<MAsset> back = mesh; // upcast — always valid
17//
18// Downcasts that don't match return nullptr from get().
19//
20// ── Serialisation ─────────────────────────────────────────────────────────────
21// toRefString() produces "guid::<id>".
22// The manager also accepts "rawp::<path>" at load time, but that is
23// converted to a GUID handle immediately. On re-save the reference is
24// always written as "guid::<id>".
25//
26// ── Creation ──────────────────────────────────────────────────────────────────
27// Handles should only be created through MAssetManager.
28// Do not construct handles manually unless you supply a valid resolver.
29// ─────────────────────────────────────────────────────────────────────────────
30

39template<typename T>
40class TAssetHandle
41{
42 static_assert(std::is_base_of_v<MAsset, T>, "T must inherit from MAsset");
43
44 // All specialisations are friends so converting constructors can
45 // read private members of TAssetHandle<U>.
46 template<typename U> friend class TAssetHandle;
47
48public:
49 // ── Resolver type ─────────────────────────────────────────────────────────
50 // Returns MAsset* (untyped) so the same function pointer works across all
51 // handle types. get() applies dynamic_cast<T*> for type safety.
52 using ResolveFunc = MAsset*(*)(const SString&);
53
54private:
55 SString assetId; // always a GUID
56 ResolveFunc resolveFunc = nullptr;
57
58public:
59 TAssetHandle() = default;
60
61 // Primary constructor — called by the asset manager when handing out handles.
62 TAssetHandle(const SString& id, ResolveFunc resolver)
63 : assetId(id), resolveFunc(resolver) {}
64
65 // ── Converting constructor ────────────────────────────────────────────────
66 // Allows conversion between any TAssetHandle<U> and TAssetHandle<T> as
67 // long as both U and T derive from MAsset.
68 //
69 // Upcast (Derived -> Base): always valid.
70 // Downcast (Base -> Derived): compiles, but get() returns nullptr if the
71 // underlying asset isn't actually a T.
72 //
73 // The resolver is copied as-is because it returns MAsset* — the
74 // dynamic_cast<T*> in get() handles type narrowing.
75 template<typename U>
76 TAssetHandle(const TAssetHandle<U>& other) // NOLINT(google-explicit-constructor)
77 : assetId(other.assetId)
78 , resolveFunc(other.resolveFunc)
79 {
80 static_assert(std::is_base_of_v<MAsset, U>, "U must inherit from MAsset");
81 }
82
83 // ── Resolution ────────────────────────────────────────────────────────────
84
85 // Resolves through the callback every call, then dynamic_cast<T*> for
86 // type safety. Returns nullptr if the handle is null, the asset isn't
87 // loaded, or the asset isn't actually a T.
88 T* get() const
89 {
90 if (assetId.empty() || !resolveFunc)
91 return nullptr;
92 return dynamic_cast<T*>(resolveFunc(assetId));
93 }
94
95 T* operator->() const { return get(); }
96 T& operator*() const { return *get(); }
97
98 // ── State queries ─────────────────────────────────────────────────────────
99
100 // True if the handle has a GUID AND the asset is currently loaded
101 // AND the asset is actually a T.
102 explicit operator bool() const { return isValid(); }
103 bool isValid() const { return get() != nullptr; }
104
105 // True if the handle was never assigned (no GUID at all).
106 bool isNull() const { return assetId.empty(); }
107
108 // ── Accessors ─────────────────────────────────────────────────────────────
109
110 const SString& getAssetId() const { return assetId; }
111
112 // ── Serialisation ─────────────────────────────────────────────────────────
113
114 // Always produces "guid::<id>". Returns empty string for null handles.
115 SString toRefString() const
116 {
117 if (assetId.empty()) return SString();
118 return SString("guid::") + assetId;
119 }
120
121 // ── Comparison ────────────────────────────────────────────────────────────
122
123 bool operator==(const TAssetHandle<T>& other) const { return assetId == other.assetId; }
124 bool operator!=(const TAssetHandle<T>& other) const { return assetId != other.assetId; }
125};
