22 // -----------------------------------------------------------------------
23 // Buffer creation
24 //
25 // Allocates a new T, calls makeBuffer(name), stores it, and returns a
26 // typed pointer. Returns nullptr if makeBuffer fails.
27 // Replaces any existing buffer registered under the same name.
28 // -----------------------------------------------------------------------
29 template<typename T>
30 T* createBuffer(const SString& name)
31 {
32 static_assert(std::is_base_of_v<SBuffer, T>,
33 "T must derive from SBuffer");
34
35 // Evict and destroy any existing registration under this name.
36 auto it = mBuffers.find(name);
37 if (it != mBuffers.end())
38 {
39 it->second->unbind();
40 delete it->second;
41 mBuffers.erase(it);
42 }
43
44 T* inst = new T();
45 if (!inst->makeBuffer(name))
46 {
47 delete inst;
48 return nullptr;
49 }
50
51 mBuffers[name] = inst;
52 return inst;
53 }
54
55 // -----------------------------------------------------------------------
56 // Buffer retrieval
57 //
58 // Returns nullptr if the name is not registered or the stored buffer
59 // cannot be cast to T.
60 // -----------------------------------------------------------------------
61 template<typename T = SBuffer>
62 T* getBuffer(const SString& name) const
63 {
64 static_assert(std::is_base_of_v<SBuffer, T>,
65 "T must derive from SBuffer");
66
67 auto it = mBuffers.find(name);
68 if (it == mBuffers.end()) return nullptr;
69 return dynamic_cast<T*>(it->second);
70 }

77 // -----------------------------------------------------------------------
78 void resizeAll(int width, int height);
