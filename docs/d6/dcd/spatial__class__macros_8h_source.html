4// ── Usage ─────────────────────────────────────────────────────────────────
5//
6// // MyEntity.h
7// class MCamera : public MSpatialEntity {
8// DEFINE_CLASS(MCamera)
9// private:
10// DECLARE_FIELD(fov, float, 90.0f)
11// };
12//
13// // MyEntity.cpp — ONE call per class, in the .cpp only, never in a header
14// IMPLEMENT_CLASS(MCamera)
15//
16// ── Why the split matters ─────────────────────────────────────────────────
17//
18// DEFINE_CLASS puts only a *declaration* of _classRegistered in the header.
19// IMPLEMENT_CLASS puts the *definition* (with the factory lambda) in the .cpp.
20//
21// At the point IMPLEMENT_CLASS expands in the .cpp the class is already
22// fully defined (its header was already included), so ClassName* → MSpatialEntity*
23// is a valid implicit conversion and the lambda compiles cleanly.
24//
25// inline static with a body-level lambda fails because the class is still
26// *incomplete* when the class body is being parsed — the compiler cannot yet
27// verify the inheritance relationship needed for the pointer conversion.
28//
29// One IMPLEMENT_CLASS per class per binary → exactly one definition of
30// _classRegistered → no ODR violations.
31//
32

39// ── DEFINE_CLASS ──────────────────────────────────────────────────────────
40// Place as the first item inside any concrete MSpatialEntity subclass body.
41// Injects typeInfo(), staticTypeInfo(), create(), and a *declaration* of
42// _classRegistered. The matching IMPLEMENT_CLASS in the .cpp provides the
43// definition.
44//
45#define DEFINE_SPATIAL_CLASS(ClassName) \
46public: \
47 static constexpr MTypeInfo staticTypeInfo() \
48 { \
49 return MAKE_TYPE_INFO(ClassName); \
50 } \
51 MTypeInfo typeInfo() const override \
52 { \
53 return staticTypeInfo(); \
54 } \
55 static ClassName* create(const SString& name = {}) \
56 { \
57 return MSpatialEntity::createInstance<ClassName>(name); \
58 } \
59private: \
60 /* Declaration only — definition is in IMPLEMENT_CLASS in the .cpp */ \
61 static const bool _classRegistered;
62
63// ── IMPLEMENT_CLASS ───────────────────────────────────────────────────────
64// Place exactly once per class in its .cpp file.
65// At this point the class is complete, so ClassName* → MSpatialEntity* is
66// valid and the factory lambda compiles without the "incomplete type" error.
67//
68#define IMPLEMENT_SPATIAL_CLASS(ClassName) \
69 const bool ClassName::_classRegistered = []() { \
70 MEntityTypeRegistry::get().registerType( \
71 ClassName::staticTypeInfo(), \
72 []() -> MSpatialEntity* { return ClassName::create(SString{}); } \
73 ); \
74 return true; \
75 }();

77// ── DEFINE_ABSTRACT_CLASS ─────────────────────────────────────────────────
78// Use for abstract intermediate classes (MLightEntity, MDynamicLight, etc.)
79// Provides typeInfo() and staticTypeInfo() but no factory and no registry
80// entry — abstract classes cannot be directly instantiated.
81//

83public: \
84 static constexpr MTypeInfo staticTypeInfo() \
85 { \

87 } \
88 MTypeInfo typeInfo() const override \
89 { \
90 return staticTypeInfo(); \
91 } \
92private:
