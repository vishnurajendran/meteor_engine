13// Owns sf::Texture* objects keyed by asset GUID.
14// Thumbnails are persisted to CACHE_DIR so they survive editor restarts
15// without needing to be re-rendered.
16//
17// ── Lifetime rules ────────────────────────────────────────────────────────────
18//
19// store() — saves to memory AND writes a PNG to CACHE_DIR/<assetId>.png
20// get() — returns from memory; on first access loads from disk if the
21// PNG exists (lazy load), so no startup scan is needed.
22// has() — true when the texture is in memory OR the disk file exists.
23// Used by requestThumbnail() to avoid re-queuing cached work.
24// evict() — removes from memory AND deletes the disk file.
25// Call this on hot-reload so the stale thumbnail is regenerated.
26// evictAll() — clears memory only. Disk files are preserved across full
27// refresh() calls because GUIDs are stable — the same asset
28// has the same ID even after a rescan.
29//
30// Thread-safety: single-threaded (editor main thread only).
31// ─────────────────────────────────────────────────────────────────────────────
32class MThumbnailCache : public MObject
33{
34 DEFINE_OBJECT_SUBCLASS(MThumbnailCache)
35public:
36 // All cache PNGs live under this directory (created automatically).
37 static constexpr const char* CACHE_DIR = ".engine_data/thumbnail_cache";

39 ~MThumbnailCache();

41 // Store a newly-generated thumbnail in memory and save it to disk.
42 // Takes ownership of the texture pointer.
43 void store(const SString& assetId, sf::Texture* texture);
44
45 // Return the texture for assetId.
46 // If not in memory, attempts to load from the disk cache.
